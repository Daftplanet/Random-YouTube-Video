<script>
    // List of YouTube URLs
    var youtubeUrls = [
        'https://www.youtube.com/watch?v=PoG-LRaf23w',
        'https://www.youtube.com/watch?v=G4La2AonVl8',
        'https://www.youtube.com/watch?v=MbnYNeuKrzc'
        // ...add as many URLs as you have
    ];

    function playRandomVideo() {
        var randomIndex = Math.floor(Math.random() * youtubeUrls.length);
        var randomUrl = youtubeUrls[randomIndex];
        var videoId = randomUrl.split('v=')[1];
        var ampersandPosition = videoId.indexOf('&');
        if(ampersandPosition != -1) {
          videoId = videoId.substring(0, ampersandPosition);
        }

        var player = document.getElementById('player');
        player.innerHTML = '<iframe width="560" height="315" src="https://www.youtube.com/embed/' + videoId + '" frameborder="0" allowfullscreen></iframe>';
    }
</script>
